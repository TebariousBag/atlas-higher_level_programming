#!/usr/bin/env python3

import unittest
from 1_my_list import MyList

class TestMyList(unittest.TestCase):
    def test_print_sorted_empty_list(self):
        my_list = MyList()
        my_list.print_sorted()
        self.assertEqual([], my_list)

    def test_print_sorted_single_element(self):
        my_list = MyList([5])
        my_list.print_sorted()
        self.assertEqual([5], my_list)

    def test_print_sorted_multiple_elements(self):
        my_list = MyList([5, 2, 8, 1, 9])
        my_list.print_sorted()
        self.assertEqual([1, 2, 5, 8, 9], my_list)

    def test_print_sorted_duplicates(self):
        my_list = MyList([5, 2, 8, 1, 9, 2, 5])
        my_list.print_sorted()
        self.assertEqual([1, 2, 2, 5, 5, 8, 9], my_list)

if __name__ == '__main__':
    unittest.main()
